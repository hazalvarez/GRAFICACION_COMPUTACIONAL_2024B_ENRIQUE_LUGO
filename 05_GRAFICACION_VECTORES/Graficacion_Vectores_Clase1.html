# Graficando el vector

x1 = -6  # Valor mnimo para el eje X
x2 = 6   # Valor mximo para el eje X

y1 = -6  # Valor mnimo para el eje Y
y2 = 6   # Valor mximo para el eje Y

# Editamos los ejes del grfico
plt.axis([x1, x2, y1, y2])

# Activamos la visualizacin de los ejes
plt.axis('on')

# Mostramos la cuadrcula en el grfico
plt.grid(True)

# Asignamos un ttulo al grfico
plt.title("EJERCICIO VECTOR B")

# Definimos la distancia entre puntos en el eje Y
dy = 5

# Definimos la distancia entre puntos en el eje X
dx = 5

# Graficar puntos en el plano
for x in np.arange(x1, x2, dx):  # Iteramos sobre el rango de valores para X
    for y in np.arange(y1, y2, dy):  # Iteramos sobre el rango de valores para Y
        # Graficamos los puntos con coordenadas (x, y)
        plt.scatter(x, y, s=1.5, color='lightgray')  # s es el tamao del punto, color es el color del punto

# Graficar vectores (flechas)
# plt.arrow(inicio_x, inicio_y, desplazamiento_x, desplazamiento_y, head_length=tamao de la punta, head_width=ancho de la punta, color=color)

# Primera flecha, desde (0, 0) con desplazamiento de 5 unidades en el eje X
plt.arrow(0, 0, 5, 0, head_length=0.3, head_width=0.3, color="k")  # Flecha negra

# Segunda flecha, desde (-2, 4) con desplazamiento de 5 unidades en el eje X
plt.arrow(-2, 4, 5, 0, head_length=0.3, head_width=0.3, color="k")  # Flecha negra

# Tercera flecha, desde (-6, -2) con desplazamiento de 5 unidades en el eje X
plt.arrow(-6, -2, 5, 0, head_length=0.3, head_width=0.3, color="k")  # Flecha negra
